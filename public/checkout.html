app.post("/checkout/get-rates", async (req, res) => { const { destinationPostalCode, country } = req.body; if (!destinationPostalCode || !country) { return res.status(400).json({ error: "Missing postal code or country" }); } const xmlRequest = ` ${process.env.CPC_USER} 0.015 N0B1M0 ${country} ${destinationPostalCode} `; try { const response = await axios.post( "https://ct.soa-gw.canadapost.ca/rs/ship/price", xmlRequest, { headers: { "Content-Type": "application/vnd.cpc.ship.rate-v4+xml", Accept: "application/vnd.cpc.ship.rate-v4+xml", }, auth: { username: process.env.CPC_USER, password: process.env.CPC_PASS, }, } ); res.type("application/xml").send(response.data); } catch (err) { console.error("Canada Post API Error:", err.message); res.status(500).json({ error: "Failed to get rates" }); } });
